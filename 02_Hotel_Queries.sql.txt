-- HOTEL QUERIES

-- Q1: Last booked room
SELECT *
FROM bookings
ORDER BY booking_date DESC
LIMIT 1;

-- Q2: Billing in Nov 2021
SELECT b.booking_id, SUM(bc.quantity * i.rate) AS total_bill
FROM bookings b
JOIN booking_commercials bc ON b.booking_id = bc.booking_id
JOIN items i ON bc.item_id = i.item_id
WHERE MONTH(b.booking_date) = 11 AND YEAR(b.booking_date) = 2021
GROUP BY b.booking_id;

-- Q3: Bills greater than 1000
SELECT b.booking_id, SUM(bc.quantity * i.rate) AS total_bill
FROM bookings b
JOIN booking_commercials bc ON b.booking_id = bc.booking_id
JOIN items i ON bc.item_id = i.item_id
GROUP BY b.booking_id
HAVING total_bill > 1000;

-- Q4: Most and Least ordered items per month
SELECT *
FROM (
    SELECT 
        MONTH(b.booking_date) AS month,
        i.item_name,
        SUM(bc.quantity) AS total_qty,
        RANK() OVER (PARTITION BY MONTH(b.booking_date) ORDER BY SUM(bc.quantity) DESC) AS max_rank,
        RANK() OVER (PARTITION BY MONTH(b.booking_date) ORDER BY SUM(bc.quantity) ASC) AS min_rank
    FROM bookings b
    JOIN booking_commercials bc ON b.booking_id = bc.booking_id
    JOIN items i ON bc.item_id = i.item_id
    GROUP BY month, i.item_name
) t
WHERE max_rank = 1 OR min_rank = 1;

-- Q5: 2nd highest bill
SELECT *
FROM (
    SELECT 
        b.booking_id,
        SUM(bc.quantity * i.rate) AS total_bill,
        DENSE_RANK() OVER (ORDER BY SUM(bc.quantity * i.rate) DESC) AS rnk
    FROM bookings b
    JOIN booking_commercials bc ON b.booking_id = bc.booking_id
    JOIN items i ON bc.item_id = i.item_id
    GROUP BY b.booking_id
) t
WHERE rnk = 2;
